<div id="history"></div>

<button onclick="location.href='/'">戻る</button>


<script>

async function load(){

  const res=await fetch("/results");

  const data=await res.json();

  const box=document.getElementById("history");


  data.forEach(r=>{

    const d=document.createElement("div");

    d.className="item";

    d.innerHTML=`

      ${r.text}<br>

      <button onclick="repost('${r.text}')">再投稿</button>

      <button onclick="edit('${r.text}')">修正</button>

    `;

    box.appendChild(d);

  });

}

load();


function repost(text){

  fetch("/post_result",{

    method:"POST",

    headers:{"Content-Type":"application/json"},

    body:JSON.stringify({text:text})

  });

  alert("再投稿しました");

}


function edit(text){

  localStorage.setItem("edit",text);

  location.href="/";

}
